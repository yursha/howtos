Loops
-----

C
~

[source,c]
....
for (int i = 0; i < 10; ++i) {
  // do something
}
....

[source,c]
....
int i = 0;
while (i < 10) {
  // do something
  ++i;
}
....

C++
~~~

[source,c++]
....
// iterators
std::vector<int> v;
for(auto const& it = v.begin(); it != v.end(); ++it) {
  std::cout << *it;
}
....

[source,c++]
....
// ranges
std::vector<int> v;
for(auto const& i: v) {
  std::cout << i;
}
....
