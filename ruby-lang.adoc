[[methods]]
methods
-------

[source, ruby]
....
def hi
  puts "Hello World!"
end

def hello(name)
  puts "Hello #{name}!"
end

def bonjur(name = "World")
  puts "Hello #{name.capitalize}!"
end

hi             # Hello World!
hello "Maria"  # Hello Maria!
bonjur         # Hello World!
bonjur "Maria" # Hello Maria!

....

[[arrays]]
arrays
------

sorting arrays
....
ticket = [34, 32, 43]
ticket.sort! // ticket == [32, 34, 43]
             // exclamation mark means sort in _in place_
             
ticket.reverse! // ticket = [43, 34, 32]
....
